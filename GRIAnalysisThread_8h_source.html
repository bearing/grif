00228   
00229   read_data_ptrs_.push_back(reinterpret_cast<void*>(p.second));
00230   
00231   // Adding NULL to p.second after this to ensure garbage collection
00232   // goes on as planned. NULL packet writes will write a single NULL

00234   // normally, but should not be passed back
00235   if (p.first == 0) {
00236     p.second = NULL;

00238   
00239   return p;
