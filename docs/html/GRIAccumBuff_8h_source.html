00038   qint64 BufferData(int numel, qint64 t[], T d[]) {
00039     qint64 dt = 0;
00040     for (int i = 0; i < numel; ++i) {
00041       if (t[i] >= t1_ && t[i] < t2_) {
00042     time_.append(t[i]);
00043     data_.append(d[i]);
00044       }
00045       // trigger on times greater than leading edge buffer
00046       if (t[i] >= t2_ && leading_edge_) {
00047         // time greater than leading edge
00048         if (t[i]-t2_ > dt) dt = t[i]-t2_;
00049       }
00050     }
00051     return dt;
00052   }
