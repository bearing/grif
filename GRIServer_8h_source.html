00075     void readAndWriteData() {
00076         QList<QTcpSocket*>::iterator itr = connected.begin();
00077 
00078         // If you are transfering a lot of data, I would suggest making this array
00079         // larger
00080         char writing[100];
00081 
00082         int read = 0;
00083 
00084         // We cycle through all the sockets and see if any of them have bytes
00085         // available to read. We do this because we don't actually know which
00086         // socket invoked this method, and we don't stop at the first one
00087         // because while we're in this function a different socket could have
00088         // written data, and we could have lost in that process.
00089         for (; itr != connected.end(); itr++) {
00090             QTcpSocket* sock = *itr;
00091             while (sock->bytesAvailable() > sizeof(char)) {
00092                 int bytesread;
00093                 if ((bytesread = sock->read((writing + read), 100 - read)) == -1) {
00094                     // If the socket we're reading from gives us bad data or its closed,
00095                     // we can safely remove it from the list, because it needs to
00096                     // reconnect
00097                     connected.removeOne(sock);
00098                     break;
00099                 }
00100                 read += bytesread;
00101             }
00102         }
00103         // Now we have all the data we need to write, so we just
00104         // cycle through everything and write it to the sockets.
00105         for (itr = connected.begin(); itr != connected.end(); itr++) {
00106             QTcpSocket* sock = *itr;
00107             sock->write(writing, read);
00108             if (!sock->waitForBytesWritten(-1)) {
00109                 std::cout << "Could NOT write buffer to socket" <<std::endl;
00110             }
00111         }
00112     }
